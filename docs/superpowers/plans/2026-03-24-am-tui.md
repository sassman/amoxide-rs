# `am-tui` Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a separate `am-tui` binary crate that provides a minimalistic ratatui-based TUI for navigating, selecting, moving, deleting, and creating aliases and profiles.

**Architecture:** Elm-architecture (model → update → view) in a new `crates/am-tui` workspace crate. The TUI wraps the existing `AppModel` from the `am` library crate. Tree state is flattened into a `Vec<TreeNode>` for rendering. Identity-based selection (`AliasId`) survives tree rebuilds.

**Tech Stack:** Rust, ratatui (with crossterm backend), `am` library crate for core types.

**Spec:** `docs/superpowers/specs/2026-03-24-tui-design.md`

```rust
impl TuiModel {
    pub fn new() -> anyhow::Result<Self> {
        let app_model = AppModel::default();
        let cwd = std::env::current_dir()?;
        let project_path = ProjectAliases::find_path(&cwd)?;
        let project_aliases = match &project_path {
            Some(path) => Some(ProjectAliases::load(path)?),
            None => None,
        };

        let mut model = Self {
            app_model,
            project_aliases,
            project_path,
            tree: Vec::new(),
            cursor: 0,
            selected: BTreeSet::new(),
            mode: Mode::Normal,
            dest_tree: Vec::new(),
            dest_cursor: 0,
            active_column: Column::Left,
            scroll_offset: 0,
        };
        model.rebuild_tree();
        Ok(model)
    }

    pub fn rebuild_tree(&mut self) {
        self.tree = crate::tree::build_tree(
            &self.app_model,
            self.project_aliases.as_ref(),
        );
        self.dest_tree = crate::tree::build_dest_tree(
            &self.app_model,
            self.project_aliases.is_some(),
        );
        // Clamp cursor to valid range
        if !self.tree.is_empty() {
            if self.cursor >= self.tree.len() {
                self.cursor = self.tree.len() - 1;
            }
            // Snap to nearest navigable node
            self.cursor = self.next_navigable(self.cursor)
                .unwrap_or(0);
        }
    }

    /// Find next navigable node at or after `from`.
    pub fn next_navigable(&self, from: usize) -> Option<usize> {
        let tree = if self.active_column == Column::Left {
            &self.tree
        } else {
            &self.dest_tree
        };
        (from..tree.len()).find(|&i| tree[i].kind.is_navigable())
    }

    /// Find previous navigable node at or before `from`.
    pub fn prev_navigable(&self, from: usize) -> Option<usize> {
        let tree = if self.active_column == Column::Left {
            &self.tree
        } else {
            &self.dest_tree
        };
        (0..=from).rev().find(|&i| tree[i].kind.is_navigable())
    }
}
```

```rust
use std::collections::BTreeMap;
use am::update::AppModel;
use am::{AliasSet, Config, Profile, ProfileConfig, ProjectAliases};
use crate::model::{AliasId, NodeKind, TreeNode};

/// Build the full flattened tree from an AppModel.
pub fn build_tree(app_model: &AppModel, project: Option<&ProjectAliases>) -> Vec<TreeNode> {
    build_tree_from_parts(
        &app_model.config.aliases,
        app_model.profile_config(),
        app_model.config.active_profile.as_deref(),
        project,
    )
}

/// Build tree from individual parts (testable without filesystem).
pub fn build_tree_from_parts(
    global_aliases: &AliasSet,
    profiles: &ProfileConfig,
    active_profile: Option<&str>,
    project: Option<&ProjectAliases>,
) -> Vec<TreeNode> {
    let mut nodes = Vec::new();

    // 1. Global aliases (if any)
    if !global_aliases.is_empty() {
        nodes.push(TreeNode {
            kind: NodeKind::GlobalHeader,
            depth: 0,
            alias_id: None,
            alias_command: None,
            is_active: false,
            label: "global".into(),
        });
        for (name, alias) in global_aliases.iter() {
            nodes.push(TreeNode {
                kind: NodeKind::AliasItem,
                depth: 1,
                alias_id: Some(AliasId::Global { alias_name: name.as_ref().to_string() }),
                alias_command: Some(alias.command().to_string()),
                is_active: false,
                label: name.as_ref().to_string(),
            });
        }
    }

    // 2. Project aliases (if any)
    if let Some(proj) = project {
        if !proj.aliases.is_empty() {
            nodes.push(TreeNode {
                kind: NodeKind::ProjectHeader,
                depth: 0,
                alias_id: None,
                alias_command: None,
                is_active: false,
                label: "project (.aliases)".into(),
            });
            for (name, alias) in proj.aliases.iter() {
                nodes.push(TreeNode {
                    kind: NodeKind::AliasItem,
                    depth: 1,
                    alias_id: Some(AliasId::Project { alias_name: name.as_ref().to_string() }),
                    alias_command: Some(alias.command().to_string()),
                    is_active: false,
                    label: name.as_ref().to_string(),
                });
            }
        }
    }

    // 3. Profile tree
    // Build parent → children mapping
    let all_profiles: Vec<&Profile> = profiles.iter().collect();
    let mut children_of: BTreeMap<&str, Vec<&Profile>> = BTreeMap::new();
    for profile in &all_profiles {
        if let Some(ref parent) = profile.inherits {
            if all_profiles.iter().any(|p| p.name == *parent) {
                children_of.entry(parent.as_str()).or_default().push(profile);
            }
        }
    }

    // Find roots
    let roots: Vec<&Profile> = all_profiles.iter()
        .filter(|p| {
            p.inherits.is_none()
                || !all_profiles.iter().any(|other| Some(&other.name) == p.inherits.as_ref())
        })
        .copied()
        .collect();

    for root in &roots {
        build_profile_nodes(root, &children_of, active_profile, 0, &mut nodes);
    }

    nodes
}

fn build_profile_nodes(
    profile: &Profile,
    children_of: &BTreeMap<&str, Vec<&Profile>>,
    active_profile: Option<&str>,
    depth: u16,
    nodes: &mut Vec<TreeNode>,
) {
    let is_active = active_profile == Some(profile.name.as_str());

    nodes.push(TreeNode {
        kind: NodeKind::ProfileHeader,
        depth,
        alias_id: None,
        alias_command: None,
        is_active,
        label: profile.name.clone(),
    });

    for (name, alias) in profile.aliases.iter() {
        nodes.push(TreeNode {
            kind: NodeKind::AliasItem,
            depth: depth + 1,
            alias_id: Some(AliasId::Profile {
                profile_name: profile.name.clone(),
                alias_name: name.as_ref().to_string(),
            }),
            alias_command: Some(alias.command().to_string()),
            is_active: false,
            label: name.as_ref().to_string(),
        });
    }

    if let Some(children) = children_of.get(profile.name.as_str()) {
        for child in children {
            build_profile_nodes(child, children_of, active_profile, depth + 1, nodes);
        }
    }
}

/// Build destination tree (headers only, same structure).
pub fn build_dest_tree(app_model: &AppModel, has_project: bool) -> Vec<TreeNode> {
    build_dest_tree_from_parts(
        &app_model.config.aliases,
        app_model.profile_config(),
        app_model.config.active_profile.as_deref(),
        has_project,
    )
}

pub fn build_dest_tree_from_parts(
    global_aliases: &AliasSet,
    profiles: &ProfileConfig,
    active_profile: Option<&str>,
    has_project: bool,
) -> Vec<TreeNode> {
    let mut nodes = Vec::new();

    // Global header (always shown in dest — you can move aliases TO global even if currently empty)
    nodes.push(TreeNode {
        kind: NodeKind::GlobalHeader,
        depth: 0,
        alias_id: None,
        alias_command: None,
        is_active: false,
        label: "global".into(),
    });

    // Project header (if project context exists, or always to allow creating .aliases)
    nodes.push(TreeNode {
        kind: NodeKind::ProjectHeader,
        depth: 0,
        alias_id: None,
        alias_command: None,
        is_active: false,
        label: "project (.aliases)".into(),
    });

    // Profile headers only (same tree structure as left column, but no aliases)
    let all_profiles: Vec<&Profile> = profiles.iter().collect();
    let mut children_of: BTreeMap<&str, Vec<&Profile>> = BTreeMap::new();
    for profile in &all_profiles {
        if let Some(ref parent) = profile.inherits {
            if all_profiles.iter().any(|p| p.name == *parent) {
                children_of.entry(parent.as_str()).or_default().push(profile);
            }
        }
    }

    let roots: Vec<&Profile> = all_profiles.iter()
        .filter(|p| {
            p.inherits.is_none()
                || !all_profiles.iter().any(|other| Some(&other.name) == p.inherits.as_ref())
        })
        .copied()
        .collect();

    for root in &roots {
        build_dest_profile_nodes(root, &children_of, active_profile, 0, &mut nodes);
    }

    nodes
}

fn build_dest_profile_nodes(
    profile: &Profile,
    children_of: &BTreeMap<&str, Vec<&Profile>>,
    active_profile: Option<&str>,
    depth: u16,
    nodes: &mut Vec<TreeNode>,
) {
    let is_active = active_profile == Some(profile.name.as_str());
    nodes.push(TreeNode {
        kind: NodeKind::ProfileHeader,
        depth,
        alias_id: None,
        alias_command: None,
        is_active,
        label: profile.name.clone(),
    });

    if let Some(children) = children_of.get(profile.name.as_str()) {
        for child in children {
            build_dest_profile_nodes(child, children_of, active_profile, depth + 1, nodes);
        }
    }
}
```

```rust
use std::collections::BTreeSet;
use crate::model::*;

pub fn update(model: &mut TuiModel, msg: TuiMessage) {
    match msg {
        TuiMessage::CursorDown => {
            let tree = active_tree(model);
            let cursor = active_cursor(model);
            if let Some(next) = (*cursor + 1..tree.len()).find(|&i| tree[i].kind.is_navigable()) {
                *active_cursor_mut(model) = next;
            }
        }
        TuiMessage::CursorUp => {
            let tree = active_tree(model);
            let cursor = active_cursor(model);
            if let Some(prev) = (0..*cursor).rev().find(|&i| tree[i].kind.is_navigable()) {
                *active_cursor_mut(model) = prev;
            }
        }
        TuiMessage::JumpTop => {
            let tree = active_tree(model);
            if let Some(first) = (0..tree.len()).find(|&i| tree[i].kind.is_navigable()) {
                *active_cursor_mut(model) = first;
            }
        }
        TuiMessage::JumpBottom => {
            let tree = active_tree(model);
            if let Some(last) = (0..tree.len()).rev().find(|&i| tree[i].kind.is_navigable()) {
                *active_cursor_mut(model) = last;
            }
        }
        TuiMessage::ToggleSelect => {
            if model.mode != Mode::Normal && model.mode != Mode::Moving {
                return;
            }
            if let Some(node) = model.tree.get(model.cursor) {
                if node.kind.is_selectable() {
                    if let Some(ref id) = node.alias_id {
                        if !model.selected.remove(id) {
                            model.selected.insert(id.clone());
                        }
                    }
                }
            }
        }
        _ => {} // other messages handled in later tasks
    }
}

    match model.active_column {
        Column::Left => &model.tree,
        Column::Right => &model.dest_tree,

        Column::Right => &model.dest_cursor,

        Column::Left => &mut model.cursor,
```

```rust
use crossterm::event::{Event, KeyCode, KeyEvent, KeyModifiers};
use crate::model::{Mode, TuiMessage};

pub fn map_event(event: &Event, mode: &Mode) -> Option<TuiMessage> {
    match event {
        Event::Key(key) => map_key(key, mode),
        Event::Resize(w, h) => Some(TuiMessage::Resize(*w, *h)),
        _ => None,
    }
}

fn map_key(key: &KeyEvent, mode: &Mode) -> Option<TuiMessage> {
    // Ctrl+C always quits
    if key.modifiers.contains(KeyModifiers::CONTROL) && key.code == KeyCode::Char('c') {
        return Some(TuiMessage::Quit);
    }

    match mode {
        Mode::Normal | Mode::Moving => map_normal_key(key, mode),
        Mode::TextInput(_) => map_text_input_key(key),
        Mode::Confirm(_) => map_confirm_key(key),
    }
}

fn map_normal_key(key: &KeyEvent, mode: &Mode) -> Option<TuiMessage> {
    match key.code {
        // Navigation
        KeyCode::Char('j') | KeyCode::Down => Some(TuiMessage::CursorDown),
        KeyCode::Char('k') | KeyCode::Up => Some(TuiMessage::CursorUp),
        KeyCode::Char('g') | KeyCode::Home => Some(TuiMessage::JumpTop),
        KeyCode::Char('G') | KeyCode::End => Some(TuiMessage::JumpBottom),

        // Selection
        KeyCode::Char(' ') => Some(TuiMessage::ToggleSelect),
        KeyCode::Char('m') => Some(TuiMessage::EnterMoveMode),
        KeyCode::Enter => Some(TuiMessage::ExecuteMove),
        KeyCode::Esc => Some(TuiMessage::CancelMove),
        KeyCode::Tab => Some(TuiMessage::SwitchColumn),

        // Management (Normal mode only)
        KeyCode::Char('n') if *mode == Mode::Normal => Some(TuiMessage::StartCreateProfile),
        KeyCode::Char('x') if *mode == Mode::Normal => Some(TuiMessage::DeleteItem),
        KeyCode::Char('s') if *mode == Mode::Normal => Some(TuiMessage::SetActive),

        // Quit
        KeyCode::Char('q') => Some(TuiMessage::Quit),

        _ => None,
    }
}

fn map_text_input_key(key: &KeyEvent) -> Option<TuiMessage> {
    match key.code {
        KeyCode::Enter => Some(TuiMessage::TextInputConfirm),
        KeyCode::Esc => Some(TuiMessage::TextInputCancel),
        KeyCode::Char(c) => Some(TuiMessage::TextInputChar(c)),
        KeyCode::Backspace => Some(TuiMessage::TextInputChar('\x08')), // handle in update
        _ => None,
    }
}

fn map_confirm_key(key: &KeyEvent) -> Option<TuiMessage> {
    match key.code {
        KeyCode::Char('y') | KeyCode::Char('Y') => Some(TuiMessage::ConfirmYes),
        KeyCode::Char('n') | KeyCode::Char('N') | KeyCode::Esc => Some(TuiMessage::ConfirmNo),
        _ => None,
    }
}
```

```rust
use ratatui::crossterm::event::{self, Event};
use ratatui::DefaultTerminal;

use crate::model::{TuiModel, TuiMessage, MIN_WIDTH, MIN_HEIGHT};

pub fn run() -> anyhow::Result<()> {
    // Check terminal size
    let (w, h) = ratatui::crossterm::terminal::size()?;
    if w < MIN_WIDTH || h < MIN_HEIGHT {
        anyhow::bail!(
            "Terminal too small ({w}x{h}). Minimum size: {MIN_WIDTH}x{MIN_HEIGHT}. Please resize and try again."
        );
    }

    let mut model = TuiModel::new()?;
    let mut terminal = ratatui::init();

    let result = run_loop(&mut terminal, &mut model);

    ratatui::restore();
    result
}

fn run_loop(terminal: &mut DefaultTerminal, model: &mut TuiModel) -> anyhow::Result<()> {
    loop {
        terminal.draw(|frame| draw(frame, model))?;

        let event = event::read()?;
        if let Some(msg) = map_event(&event, &model.mode) {
            if msg == TuiMessage::Quit {
                break;
            }
            if let TuiMessage::Resize(w, h) = msg {
                if w < MIN_WIDTH || h < MIN_HEIGHT {
                    break; // exit on too-small resize
                }
                continue;
            }
            update(model, msg);
        }
    }
    Ok(())
}
```

```rust
use ratatui::widgets::Paragraph;
use ratatui::Frame;

pub fn draw(frame: &mut Frame, model: &TuiModel) {
    let area = frame.area();

    // Help bar at top
    let help_text = help_bar_text(&model.mode);
    let help = Paragraph::new(help_text).style(Style::default().fg(Color::DarkGray));

    // Split: help bar (1 line) + main content
    let chunks = Layout::default()
        .direction(Direction::Vertical)
        .constraints([Constraint::Length(1), Constraint::Min(0)])
        .split(area);

    frame.render_widget(help, chunks[0]);

    // Render tree
    let tree_lines = render_tree_lines(model);
    let visible_height = chunks[1].height as usize;

    // Adjust scroll_offset so cursor is visible
    // (mutability handled outside draw — model.scroll_offset set in update)

    let start = model.scroll_offset;
    let end = (start + visible_height).min(tree_lines.len());
    let visible: Vec<Line> = tree_lines[start..end].to_vec();

    let tree_widget = Paragraph::new(Text::from(visible));
    frame.render_widget(tree_widget, chunks[1]);
}

fn render_tree_lines(model: &TuiModel) -> Vec<Line<'static>> {
    let mut lines = Vec::new();

    for (i, node) in model.tree.iter().enumerate() {
        let is_cursor = i == model.cursor && model.active_column == Column::Left;
        let is_selected = node.alias_id.as_ref().is_some_and(|id| model.selected.contains(id));

        let indent = "  ".repeat(node.depth as usize);

        match &node.kind {
            NodeKind::GlobalHeader => {
                let marker = if is_cursor { "▸ " } else { "  " };
                lines.push(Line::from(vec![
                    Span::raw(format!("{indent}{marker}")),
                    Span::raw("🌐 "),
                    Span::styled("global", Style::default().bold()),
                ]));
            }
            NodeKind::ProjectHeader => {
                let marker = if is_cursor { "▸ " } else { "  " };
                lines.push(Line::from(vec![
                    Span::raw(format!("{indent}{marker}")),
                    Span::raw("📁 "),
                    Span::styled("project (.aliases)", Style::default().bold()),
                ]));
            }
            NodeKind::ProfileHeader => {
                let icon = if node.is_active { "●" } else { "○" };
                let marker = if is_cursor { "▸ " } else { "  " };
                let active_tag = if node.is_active { " (active)" } else { "" };
                lines.push(Line::from(vec![
                    Span::raw(format!("{indent}{marker}")),
                    Span::styled(
                        format!("{icon} {}{active_tag}", node.label),
                        Style::default().bold(),
                    ),
                ]));
            }
            NodeKind::AliasItem => {
                let marker = if is_cursor {
                    "▸ "
                } else if is_selected {
                    "■ "
                } else {
                    "  "
                };

                let name_style = if is_selected {
                    Style::default().fg(Color::Yellow)
                } else if is_cursor {
                    Style::default().fg(Color::White)
                } else {
                    Style::default()
                };

                // Alias name line
                lines.push(Line::from(vec![
                    Span::raw(format!("{indent}  {marker}")),
                    Span::styled(node.label.clone(), name_style),
                ]));

                // Alias command line (dimmed)
                if let Some(ref cmd) = node.alias_command {
                    lines.push(Line::from(vec![
                        Span::raw(format!("{indent}    ")),
                        Span::styled(cmd.clone(), Style::default().fg(Color::DarkGray)),
                    ]));
                }

                // Blank line after each alias for spacing
                lines.push(Line::from(""));
            }
        }
    }

    lines
}
```

```rust
fn execute_move(model: &mut TuiModel) {
    let dest_node = match model.dest_tree.get(model.dest_cursor) {
        Some(node) => node.clone(),
        None => return,
    };

    let dest_scope = match dest_node.kind {
        NodeKind::GlobalHeader => MoveDestination::Global,
        NodeKind::ProjectHeader => MoveDestination::Project,
        NodeKind::ProfileHeader => MoveDestination::Profile(dest_node.label.clone()),
        _ => return,
    };

    // Collect selected aliases
    let selected: Vec<AliasId> = model.selected.iter().cloned().collect();

    // Check for name collisions
    // ... (collision detection, potentially entering Confirm mode)

    // Execute the moves
    for alias_id in &selected {
        move_single_alias(model, alias_id, &dest_scope);
    }

    // Persist

    // Reset state
    model.selected.clear();

enum MoveDestination {
    Global,
    Project,
    Profile(String),
}

fn move_single_alias(model: &mut TuiModel, alias_id: &AliasId, dest: &MoveDestination) {
    // 1. Read the alias data from source
    let (alias_name, alias_value) = match alias_id {
        AliasId::Global { alias_name } => {
            let key = am::AliasName::from(alias_name.as_str());
            match model.app_model.config.aliases.get(&key) {
                Some(val) => (alias_name.clone(), val.clone()),
                None => return,
            }
        }
        AliasId::Profile { profile_name, alias_name } => {
            let key = am::AliasName::from(alias_name.as_str());
            match model.app_model.profile_config()
                .get_profile_by_name(profile_name)
                .and_then(|p| p.aliases.get(&key))
            {
                Some(val) => (alias_name.clone(), val.clone()),
                None => return,
            }
        }
        AliasId::Project { alias_name } => {
            let key = am::AliasName::from(alias_name.as_str());
            match model.project_aliases.as_ref().and_then(|p| p.aliases.get(&key)) {
                Some(val) => (alias_name.clone(), val.clone()),
                None => return,
            }
        }
    };

    // 2. Check if source == destination (no-op)
    let is_same = match (alias_id, dest) {
        (AliasId::Global { .. }, MoveDestination::Global) => true,
        (AliasId::Project { .. }, MoveDestination::Project) => true,
        (AliasId::Profile { profile_name, .. }, MoveDestination::Profile(dest_name)) => profile_name == dest_name,
        _ => false,
    };
    if is_same { return; }

    // 3. Remove from source
    let key = am::AliasName::from(alias_name.as_str());
    match alias_id {
        AliasId::Global { .. } => { let _ = model.app_model.config.remove_alias(&alias_name); }
        AliasId::Profile { profile_name, .. } => {
            if let Some(p) = model.app_model.profile_config_mut().get_profile_by_name_mut(profile_name) {
                let _ = p.remove_alias(&alias_name);
            }
        }
        AliasId::Project { .. } => {
            if let Some(ref mut p) = model.project_aliases {
                p.aliases.remove(&key);
            }
        }
    }

    // 4. Add to destination
    let cmd = alias_value.command().to_string();
    let raw = matches!(&alias_value, am::TomlAlias::Detailed(d) if d.raw);
    match dest {
        MoveDestination::Global => {
            model.app_model.config.add_alias(alias_name, cmd, raw);
        }
        MoveDestination::Profile(name) => {
            if let Some(p) = model.app_model.profile_config_mut().get_profile_by_name_mut(name) {
                let _ = p.add_alias(alias_name, cmd, raw);
            }
        }
        MoveDestination::Project => {
            let project = model.project_aliases.get_or_insert_with(ProjectAliases::default);
            project.add_alias(alias_name, cmd, raw);
            // Create .aliases path if needed
            if model.project_path.is_none() {
                model.project_path = Some(std::env::current_dir().unwrap().join(".aliases"));
            }
        }
    }
}

fn save_all(model: &TuiModel) {

    let _ = model.app_model.profile_config().save();
    if let (Some(ref project), Some(ref path)) = (&model.project_aliases, &model.project_path) {
        let _ = project.save(path);
    }
}
```

```rust
fn delete_alias(model: &mut TuiModel, alias_id: &AliasId) {
    match alias_id {
        AliasId::Global { alias_name } => {
            let _ = model.app_model.config.remove_alias(alias_name);
        }
        AliasId::Profile { profile_name, alias_name } => {
            if let Some(p) = model.app_model.profile_config_mut().get_profile_by_name_mut(profile_name) {
                let _ = p.remove_alias(alias_name);
            }
        }

                let key = am::AliasName::from(alias_name.as_str());
                p.aliases.remove(&key);
            }
        }
    }
    model.selected.remove(alias_id);
}
```

```rust
fn render_right_column(frame: &mut Frame, model: &TuiModel, area: Rect) {
    let mut lines = Vec::new();

    // Arrow indicator
    lines.push(Line::from(Span::styled(
        "  ──────────────────►",
        Style::default().fg(Color::DarkGray),
    )));
    lines.push(Line::from(""));

    for (i, node) in model.dest_tree.iter().enumerate() {
        let is_cursor = i == model.dest_cursor && model.active_column == Column::Right;
        let indent = "  ".repeat(node.depth as usize);
        let marker = if is_cursor { "▸ " } else { "  " };

        let line = match &node.kind {
            NodeKind::GlobalHeader => Line::from(vec![
                Span::raw(format!("{indent}{marker}")),
                Span::raw("🌐 "),
                Span::styled("global", Style::default().bold()),
            ]),
            NodeKind::ProjectHeader => Line::from(vec![
                Span::raw(format!("{indent}{marker}")),
                Span::raw("📁 "),
                Span::styled("project (.aliases)", Style::default().bold()),
            ]),
            NodeKind::ProfileHeader => {
                let icon = if node.is_active { "●" } else { "○" };
                let active_tag = if node.is_active { " (active)" } else { "" };
                Line::from(vec![
                    Span::raw(format!("{indent}{marker}")),
                    Span::styled(
                        format!("{icon} {}{active_tag}", node.label),
                        Style::default().bold(),
                    ),
                ])
            }
            _ => continue,
        };
        lines.push(line);
    }

    let widget = Paragraph::new(Text::from(lines));
    frame.render_widget(widget, area);
}
```

```rust
impl TuiModel {
    pub fn adjust_scroll(&mut self, visible_height: usize) {
        // Each alias takes ~3 lines (name + command + blank), headers ~1 line
        // Use the raw cursor index as an approximation for line position
        // A more precise approach: count actual rendered lines up to cursor
        let cursor_line = self.estimate_line_for_cursor();

        if cursor_line < self.scroll_offset {
            self.scroll_offset = cursor_line;
        } else if cursor_line >= self.scroll_offset + visible_height {
            self.scroll_offset = cursor_line - visible_height + 1;
        }
    }

    fn estimate_line_for_cursor(&self) -> usize {
        // Count rendered lines for all nodes before cursor
        let mut line = 0;
        for (i, node) in self.tree.iter().enumerate() {
            if i == self.cursor { break; }
            line += match node.kind {
                NodeKind::AliasItem => 3, // name + command + blank
                _ => 1, // headers
            };
        }
        line
    }
}
```
